96. Unique Binary Search Trees â€” Medium

Given an integer n, return the number of structurally unique BST' s (binary search trees) which has exactly n nodes of unique values from 1 to n.

Example 1:

Input: n = 3
Output: 5

Example 2:

Input: n = 1
Output: 1

Constraints:

  • 1 <= n <= 19

class Solution {
public:
vector<int> visited;
int numTrees(int n) {
// assign vector
visited.assign(n + 1, 0);
// if 0 node: 1(since will be multiplied with dp(n - i - 1))
visited[0] = 1;
// for n nodes
if (n <= 1) return 1;
return dp(n);
}
int dp(int n) {
// last node(boundary)
if (visited[n]) return visited[n];
// record current answer(number of unique child trees)
int ans = 0;
for (int i = 0; i < n; ++i) {
// for each child node, get number of unique tress it can have
ans += dp(i) * dp(n - i - 1);
}
return visited[n] = ans;
}
};
